        const app = {

            data: {

                articles: [],

                categories: [

                    { id: 'tech', name: 'تقنية', icon: '💻' },

                    { id: 'edu', name: 'تعليم', icon: '📚' },

                    { id: 'health', name: 'صحة', icon: '❤️' },

                    { id: 'travel', name: 'سفر', icon: '✈️' },

                    { id: 'food', name: 'طعام', icon: '🍽️' }

                ],

                currentArticle: null

            init() {

                this.loadData();

                this.renderArticles();

                this.renderCategories();

                this.updateStats();

                this.populateCategorySelects();

                // Keyboard shortcuts

                document.addEventListener('keydown', (e) => {

                    if (e.ctrlKey && e.key === 's') {

                        e.preventDefault();

                        if (document.getElementById('view-editor').classList.contains('active')) {

                            this.saveArticle();

                        }

                    }

                });

            },

            loadData() {

                try {

                    const saved = localStorage.getItem('articles_app_data');

                    if (saved) {

                        const parsed = JSON.parse(saved);

                        this.data.articles = parsed.articles || [];

                        if (parsed.categories) this.data.categories = parsed.categories;

                    }

                } catch (e) {

                    console.error('Error loading data:', e);

                }

            },

            saveData() {

                try {

                    localStorage.setItem('articles_app_data', JSON.stringify(this.data));

                } catch (e) {

                    this.showToast('⚠️ خطأ في حفظ البيانات', 'error');

                }

            },

            switchView(viewName) {

                document.querySelectorAll('.view').forEach(v => v.classList.remove('active'));

                document.getElementById(`view-${viewName}`).classList.add('active');

                document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));

                if (viewName === 'home') document.querySelectorAll('.nav-item')[0].classList.add('active');

                if (viewName === 'categories') document.querySelectorAll('.nav-item')[1].classList.add('active');

                // Show/hide FAB

                const fab = document.getElementById('fab');

                fab.style.display = viewName === 'home' ? 'flex' : 'none';

                window.scrollTo(0, 0);

            },

            goHome() {

                this.switchView('home');

                this.renderArticles();

                this.updateStats();

            },

            newArticle() {

                document.getElementById('article-id').value = '';

                document.getElementById('article-title').value = '';

                document.getElementById('article-category').value = '';

                document.getElementById('article-content').value = '';

                this.switchView('editor');

                document.getElementById('article-title').focus();

            },

            editArticle(id) {

                const article = this.data.articles.find(a => a.id === id);

                if (!article) return;

                document.getElementById('article-id').value = article.id;

                document.getElementById('article-title').value = article.title;

                document.getElementById('article-category').value = article.categoryId || '';

                document.getElementById('article-content').value = article.content;

                this.switchView('editor');

            },

            editCurrentArticle() {

                if (this.data.currentArticle) {

                    this.editArticle(this.data.currentArticle.id);

                }

            },

            saveArticle() {

                const id = document.getElementById('article-id').value;

                const title = document.getElementById('article-title').value.trim();

                const categoryId = document.getElementById('article-category').value;

                const content = document.getElementById('article-content').value.trim();

                if (!title || !content) {

                    this.showToast('⚠️ يرجى ملء جميع الحقول المطلوبة');

                    return;

                }

                const category = this.data.categories.find(c => c.id === categoryId);

                if (id) {

                    const index = this.data.articles.findIndex(a => a.id === id);

                    if (index !== -1) {

                        this.data.articles[index] = {

                            ...this.data.articles[index],

                            title,

                            categoryId,

                            categoryName: category ? category.name : '',

                            content,

                            updatedAt: new Date().toISOString()

                        };

                        this.showToast('✅ تم تحديث المقالة بنجاح');

                    }

                } else {

                    const newArticle = {

                        id: 'art_' + Date.now(),

                        title,

                        categoryId,

                        categoryName: category ? category.name : '',

                        content,

                        createdAt: new Date().toISOString(),

                        updatedAt: new Date().toISOString()

                    };

                    this.data.articles.unshift(newArticle);

                    this.showToast('✅ تم إنشاء المقالة بنجاح');

                }

                this.saveData();

                this.goHome();

            },

            deleteArticle(id) {

                this.showModal(

                    '🗑️ حذف المقالة',

                    'هل أنت متأكد من حذف هذه المقالة؟ لا يمكن التراجع عن هذا الإجراء.',

                    [

                        { text: 'حذف', class: 'btn-danger', action: () => {

                            this.data.articles = this.data.articles.filter(a => a.id !== id);

                            this.saveData();

                            this.renderArticles();

                            this.updateStats();

                            this.showToast('🗑️ تم حذف المقالة');

                            this.closeModal();

                        }},

                        { text: 'إلغاء', class: 'btn-secondary', action: () => this.closeModal() }

                    ]

                );

            },

            readArticle(id) {

                const article = this.data.articles.find(a => a.id === id);

                if (!article) return;

                this.data.currentArticle = article;

                document.getElementById('reader-title').textContent = article.title;

                document.getElementById('reader-category').textContent = article.categoryName || 'عام';

                document.getElementById('reader-date').textContent = '📅 ' + this.formatDate(article.createdAt);

                document.getElementById('reader-words').textContent = '📝 ' + article.content.split(/\s+/).length + ' كلمة';

                document.getElementById('reader-content').textContent = article.content;

                this.switchView('reader');

            },

            renderArticles() {

                const search = document.getElementById('search-input').value.toLowerCase();

                const filterCat = document.getElementById('filter-category').value;

                let articles = this.data.articles;

                if (search) {

                    articles = articles.filter(a =>

                        a.title.toLowerCase().includes(search) ||

                        a.content.toLowerCase().includes(search)

                    );

                }

                if (filterCat) {

                    articles = articles.filter(a => a.categoryId === filterCat);

                }

                const container = document.getElementById('articles-list');

                const emptyState = document.getElementById('empty-state');

                const countLabel = document.getElementById('articles-count');

                countLabel.textContent = `${articles.length} مقالة`;

                if (articles.length === 0) {

                    container.innerHTML = '';

                    emptyState.style.display = 'block';

                    return;

                }

                emptyState.style.display = 'none';

                container.innerHTML = articles.map(article => `

                    <div class="article-card" onclick="app.readArticle('${article.id}')">

                        <div class="article-header">

                            <div>

                                ${article.categoryName ? `<span class="article-category">${article.categoryName}</span>` : ''}

                                <h3 class="article-title">${this.escapeHtml(article.title)}</h3>

                            </div>

                        </div>

                        <p class="article-excerpt">${this.escapeHtml(article.content.substring(0, 150))}...</p>

                        <div class="article-meta">

                            <span>${this.formatDate(article.createdAt)}</span>

                            <div class="article-actions" onclick="event.stopPropagation()">

                                <button class="edit-btn" onclick="app.editArticle('${article.id}')">✏️</button>

                                <button class="delete-btn" onclick="app.deleteArticle('${article.id}')">🗑️</button>

                            </div>

                        </div>

                    </div>

                `).join('');

            },

            renderCategories() {

                const grid = document.getElementById('categories-grid');

                grid.innerHTML = this.data.categories.map(cat => {

                    const count = this.data.articles.filter(a => a.categoryId === cat.id).length;

                    return `

                        <div class="category-card" onclick="app.filterByCategory('${cat.id}')">

                            <div class="category-icon">${cat.icon}</div>

                            <div class="category-name">${cat.name}</div>

                            <div class="category-count">${count} مقالة</div>

                        </div>

                    `;

                }).join('');

            },

            filterByCategory(catId) {

                document.getElementById('filter-category').value = catId;

                this.switchView('home');

                this.renderArticles();

                document.querySelectorAll('.nav-item')[0].classList.add('active');

            },

            populateCategorySelects() {

                const selects = [document.getElementById('article-category'), document.getElementById('filter-category')];

                selects.forEach(select => {

                    const currentValue = select.value;

                    const isFilter = select.id === 'filter-category';

                    select.innerHTML = isFilter ? '<option value="">جميع الأقسام</option>' : '<option value="">اختر قسماً</option>';

                    this.data.categories.forEach(cat => {

                        const option = document.createElement('option');

                        option.value = cat.id;

                        option.textContent = cat.icon + ' ' + cat.name;

                        select.appendChild(option);

                    });

                    if (currentValue) select.value = currentValue;

                });

            },

            showAddCategory() {

                this.showModal(

                    '📂 قسم جديد',

                            this.closeModal();

                        }},

                        { text: 'إلغاء', class: 'btn-secondary', action: () => this.closeModal() }

                    ]

                );

                setTimeout(() => document.getElementById('new-cat-name').focus(), 100);

            },

            updateStats() {

                document.getElementById('stat-total').textContent = this.data.articles.length;

                document.getElementById('stat-categories').textContent = this.data.categories.length;

                const now = new Date();

                const thisMonth = this.data.articles.filter(a => {

                    const d = new Date(a.createdAt);

                    return d.getMonth() === now.getMonth() && d.getFullYear() === now.getFullYear();

                }).length;

                document.getElementById('stat-this-month').textContent = thisMonth;

            },

            insertTag(before, after) {

                const textarea = document.getElementById('article-content');

                const start = textarea.selectionStart;

                const end = textarea.selectionEnd;

                const text = textarea.value;

                const selected = text.substring(start, end);

                textarea.value = text.substring(0, start) + before + selected + after + text.substring(end);

                textarea.focus();

                textarea.setSelectionRange(start + before.length, start + before.length + selected.length);

            },

            formatDate(isoString) {

                if (!isoString) return '';

                const date = new Date(isoString);

                return date.toLocaleDateString('ar-SA', { year: 'numeric', month: 'short', day: 'numeric' });

            },

            escapeHtml(text) {

                const div = document.createElement('div');

                div.textContent = text;

                return div.innerHTML;

            },

            showToast(message) {

                const container = document.getElementById('toast-container');

                const toast = document.createElement('div');

                toast.className = 'toast';

                toast.textContent = message;

                container.appendChild(toast);

                setTimeout(() => {

                    toast.style.opacity = '0';

                    toast.style.transform = 'translateY(-20px)';

                    setTimeout(() => toast.remove(), 300);

                }, 2500);

            },

            showModal(title, body, actions) {

                document.getElementById('modal-title').textContent = title;

                document.getElementById('modal-body').innerHTML = body;

                const actionsContainer = document.getElementById('modal-actions');

                actionsContainer.innerHTML = actions.map(a =>

                    `<button class="btn ${a.class}" onclick="(${a.action.toString()})()">${a.text}</button>`

                ).join('');

                document.getElementById('modal').classList.add('active');

            },

            closeModal() {

                document.getElementById('modal').classList.remove('active');

            },

            exportData() {

                const dataStr = JSON.stringify(this.data, null, 2);

                const blob = new Blob([dataStr], { type: 'application/json' });

                const url = URL.createObjectURL(blob);

                const a = document.createElement('a');

                a.href = url;

                a.download = `articles_backup_${new Date().toISOString().split('T')[0]}.json`;

                a.click();

                URL.revokeObjectURL(url);

                this.showToast('📥 تم تصدير البيانات');

            },

            importData() {

                const input = document.createElement('input');

                input.type = 'file';

                input.accept = '.json';

                input.onchange = (e) => {

                    const file = e.target.files[0];

                    if (!file) return;

                    const reader = new FileReader();

                    reader.onload = (event) => {

                        try {

                            const data = JSON.parse(event.target.result);

                            if (data.articles) this.data.articles = data.articles;

                            if (data.categories) this.data.categories = data.categories;

                            this.saveData();

                            this.renderArticles();

                            this.renderCategories();

                            this.updateStats();

                            this.populateCategorySelects();

                            this.showToast('📤 تم استيراد البيانات بنجاح');

                        } catch (err) {

                            this.showToast('⚠️ ملف غير صالح');

                        }

                    };

                    reader.readAsText(file);

                };

                input.click();

            }

        };

        // Initialize

        document.addEventListener('DOMContentLoaded', () => app.init());

    </script>
